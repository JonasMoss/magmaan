\section*{Purpose}

This note fixes the algebra for extending \texttt{robust\_weighted\_moment\_ij}
beyond complete all-ordinal ULS/DWLS/WLS. The target is the actual estimator whose
second-stage weight is built from the data. The companion note
\path{misspec_observed_bread.tex} derives the observed-Hessian bread and the
complete all-ordinal diagonal-weight case. Here the goal is the general formula,
the full-weight formula, and the implementation grid.

\section{Moment estimator and scale}

Let $\hat\kappa$ be a vector of stage-1 moments with casewise expansion
\[
  \hat\kappa = \kappa_0 + \frac1N\sum_{i=1}^N g_i + o_p(N^{-1/2}), \qquad
  \frac1N\sum_i g_i = O_p(N^{-1/2}).
\]
For disjoint groups or blocks, stack the block moment vectors and set rows to
zero outside the block. The model-implied moments are $\sigma(\theta)$,
$d(\theta)=\sigma(\theta)-\hat\kappa$, and
$\Delta=\partial\sigma/\partial\theta\T$ is $m\times q$. Equality constraints are
represented by a full-rank map $K$ from reduced coordinates $\alpha$ to full
$\theta$, and $D=\Delta K$.

The fitted value minimizes
\[
  F_N(\theta)=\frac12 d(\theta)\T \hat W d(\theta),
\]
or the corresponding block sum. The reduced estimating equation is
\[
  U_N(\alpha)=D(\alpha)\T \hat W d(\alpha)=0.
\]
At the pseudo-true value, let
\[
  A = \left.\frac{\partial U(\alpha)}{\partial\alpha\T}\right|_{\alpha_*}
\]
be the observed-Hessian bread. This includes the residual-contracted curvature
term from $d(\theta)$; it is not the Gauss-Newton bread except under correct
specification.

\subsection{Continuous-LS observed bread}

For complete continuous LS, write the block moment residual as
\[
  d_b(\theta)=
  \begin{bmatrix}
    \mu_b(\theta)-\bar x_b\\
    \vech\{\Sigma_b(\theta)-S_b\}
  \end{bmatrix},\qquad
  \Delta_b=\frac{\partial d_b(\theta)}{\partial\theta\T},\qquad
  \pi_b=\frac{n_b}{N}.
\]
The per-unit objective is
\[
  F_N(\theta)=\frac12\sum_b \pi_b d_b(\theta)\T W_b d_b(\theta),
\]
with $W_b$ treated as fixed when differentiating the bread. Hence the
full-parameter observed Hessian is
\begin{equation}\label{eq:continuous-bread}
  H_{\theta,ac}
  =
  \sum_b \pi_b
  \left[
    \Delta_{b,a}\T W_b\Delta_{b,c}
    + h_{b,\mu}\T
      \frac{\partial^2\mu_b}{\partial\theta_a\partial\theta_c}
    + h_{b,\Sigma}\T
      \vech\left\{
      \frac{\partial^2\Sigma_b}
           {\partial\theta_a\partial\theta_c}\right\}
  \right],
  \qquad h_b=W_bd_b .
\end{equation}
The reduced bread used by the IJ is $A=K\T H_\theta K$. Implementation computes
the covariance curvature term through the LISREL trace contraction already used
by analytic observed-information paths. If $x=h_{b,\Sigma}$ is in lower-vech
order, the symmetric trace weight $G_b$ is defined by
$G_{rr}=x_{rr}$ and $G_{rc}=G_{cr}=x_{rc}/2$ for $r\ne c$, so that
$\tr(G_b X)=x\T\vech(X)$ for every symmetric $X$. Then the last term in
\eqref{eq:continuous-bread} is
$\tr\{G_b\,\partial^2\Sigma_b/
(\partial\theta_a\partial\theta_c)\}$.
The finite-difference bread remains the validation gate for
\eqref{eq:continuous-bread}; it is no longer the production path for complete
continuous LS.

\section{General estimated-weight IJ}

Assume the weight has the first-order expansion
\[
  \hat W = W + \frac1N\sum_{i=1}^N \dot W_i + o_p(N^{-1/2}),
  \qquad \sum_i \dot W_i = O_p(N^{1/2}),
\]
where $\dot W_i=\IF_i(\hat W)$. Perturbing one case gives
\[
  \delta_i d = -g_i, \qquad
  \delta_i U = D\T(\dot W_i d - Wg_i).
\]
Define the moment-space row contribution
\begin{equation}\label{eq:v}
  v_i = Wg_i - \dot W_i d.
\end{equation}
Then
\begin{theorem}[estimated-weight IJ]\label{thm:ij}
For the data-weighted moment-quadratic estimator,
\[
  \IF_i(\hat\alpha) = A^{-1}D\T v_i,
\]
up to an immaterial global sign convention for $g_i$. Hence
\begin{equation}\label{eq:vcov}
  \widehat{\Var}(\hat\theta)
  =
  K A^{-1}
    \left\{ \frac1{N^2}\sum_{i=1}^N D\T v_i v_i\T D \right\}
  A^{-1}K\T .
\end{equation}
\end{theorem}

\begin{proof}
Linearize $0=U_N(\hat\alpha)$ around $\alpha_*$. The data perturbation is
$D\T(\dot W_i d-Wg_i)$ and the parameter perturbation is
$A(\hat\alpha-\alpha_*)$. Solving for $\hat\alpha-\alpha_*$ yields
$A^{-1}D\T(Wg_i-\dot W_i d)/N$ per case. Summing the outer products gives
\eqref{eq:vcov}.
\end{proof}

\begin{corollary}[fixed-weight reduction]\label{cor:fixed}
If $\dot W_i=0$, $v_i=Wg_i$. Since
$\hat\Gamma=N^{-1}\sum_i g_i g_i\T$, the IJ covariance in \eqref{eq:vcov}
reduces exactly to
\[
  K A^{-1}D\T W\hat\Gamma W D A^{-1}K\T / N.
\]
\end{corollary}

This is the primary regression gate: ULS and any caller-declared fixed weight
must be machine-identical to the existing observed-bread sandwich.

\section{Weight derivatives}

\subsection{Diagonal empirical weight}

Let $\hat W=\diag(\hat\gamma)^{-1}$ where
$\hat\gamma=\diag(\hat\Gamma)$. Then
\[
  \dot W_i = -\diag(\gamma^{-2}\odot \dot\gamma_i),
  \qquad
  v_i = Wg_i + \diag(d\oslash \gamma^2)\dot\gamma_i .
\]
In row form this is the implemented all-ordinal DWLS correction:
\[
  v_{ik}=g_{ik}/\gamma_k + d_k\,\dot\gamma_{ik}/\gamma_k^2 .
\]
For complete continuous DWLS, $\dot\gamma_i$ is the diagonal of
\eqref{eq:continuous-adf-gamma-if}. The implementation deliberately discards
the off-diagonal empirical-Gamma influence before differentiating
$\diag(\hat\Gamma)^{-1}$; those off-diagonal channels belong to full WLS.

\subsection{Full empirical weight}

Let $\hat W=\hat\Gamma^{-1}$. Matrix inverse differentiation gives
\[
  \dot W_i=-W\dot\Gamma_i W,
  \qquad
  v_i = Wg_i + W\dot\Gamma_i Wd .
\]
This is the full-WLS target. The required ingredient is the full matrix influence
$\dot\Gamma_i$, not only its diagonal. A correct implementation must preserve
symmetry and include all channels by which the stage-1 sandwich changes.

For complete continuous WLS/ADF, write $x_i=y_i-\bar y$ and
$c_i=\vech(x_ix_i\T)-\vech(S_N)$. The moment-influence row is
$z_i=c_i$ without meanstructure and $z_i=(x_i\T,c_i\T)\T$ with
meanstructure, so
\[
  \hat\Gamma = \frac1N\sum_{j=1}^N z_jz_j\T .
\]
Under an infinitesimal case-weight perturbation at case $i$, set $h=x_i$.
The exact finite-sample derivative is
\begin{equation}\label{eq:continuous-adf-gamma-if}
  \dot\Gamma_i
  =
  z_i z_i\T - \hat\Gamma
  + \frac1N\sum_{j=1}^N
      \left\{\dot z_j(h) z_j\T + z_j\dot z_j(h)\T\right\}.
\end{equation}
If mean rows are present, $\dot z_{j,\mu}(h)=-h$. For the covariance
component indexed by the lower-triangle pair $(a,b)$,
\[
  \dot z_{j,ab}(h)=-(h_a x_{jb}+x_{ja}h_b).
\]
The derivative of the subtracted covariance $S_N$ is constant across $j$ and
therefore drops from \eqref{eq:continuous-adf-gamma-if} because
$N^{-1}\sum_j z_j=0$. The implemented continuous WLS correction uses
$d\T W\dot\Gamma_iW$ in row form and is regression-tested against central
finite differences of the contaminated empirical weight
$\hat W_\epsilon=\hat\Gamma_\epsilon^{-1}$, including the stacked
mean--covariance cross-block.

\subsection{DLS and mixed weights}

For a Browne-style mixture
\[
  \hat W = \left((1-a)\hat\Gamma_0 + a\hat\Gamma_1\right)^{-1},
\]
write $M=(1-a)\Gamma_0+a\Gamma_1$. Then
\[
  \dot W_i = -W\{(1-a)\dot\Gamma_{0i}+a\dot\Gamma_{1i}\}W,
  \qquad
  v_i = Wg_i + W\{(1-a)\dot\Gamma_{0i}+a\dot\Gamma_{1i}\}Wd .
\]
If $\Gamma_0$ is model-implied and depends only on $\theta$, its derivative
belongs in the bread. If $\Gamma_0$ is built from saturated sample moments, as
in two-stage FIML's normal-theory Stage-2 weight, its casewise derivative is a
weight-IJ term.
For complete-data continuous DLS with caller-fixed $a$,
$\Gamma_0=\Gamma_{\mathrm{NT}}(\hat S)$ and
$\Gamma_1=\hat\Gamma_{\mathrm{ADF}}$. The implemented row correction is the
same convex mixture of the normal-theory and dense empirical-Gamma IF channels,
evaluated with the DLS weight. In meanstructure layouts the normal-theory side
is $\operatorname{blockdiag}(\hat S,\Gamma_{\mathrm{NT}}(\hat S))$ and the
empirical side is the full stacked Browne $\hat\Gamma_{\mathrm{ADF}}$, including the
mean--covariance cross-block. The scalar $a$ is treated as fixed; an
empirical-Bayes rule for selecting $a$ would add a separate scalar IF term.

\section{Stage-1 sandwich influence}

Many magmaan moments are themselves M-estimators. A common form is
\[
  \hat\Gamma = \hat A^{-1}\hat V\hat A^{-\T},
  \qquad
  \hat V=N^{-1}\sum_i s_i s_i\T,\qquad
  \hat A=N^{-1}\sum_i b_i .
\]
The data-direct influence at fixed stage-1 parameter is
\begin{align*}
  \dot\Gamma_i^{\mathrm{direct}}
  &=
  A^{-1}(s_is_i\T-V)A^{-\T}
  - A^{-1}(b_i-A)\Gamma
  - \Gamma(b_i-A)\T A^{-\T}.
\end{align*}
If $\Gamma$ also changes through the movement of the stage-1 parameter
$\kappa$, add
\[
  \dot\Gamma_i^{\kappa} =
  \left[\frac{\partial \Gamma}{\partial\kappa}\right] g_i .
\]
The all-ordinal path now computes
$\dot\Gamma_i^{\mathrm{direct}}$ from integer data and the diagonal of
$\dot\Gamma_i^\kappa$ by finite differences for DWLS. Full WLS uses the dense
matrix $\dot\Gamma_i^{\mathrm{direct}}+\dot\Gamma_i^\kappa$ with the row
correction $d\T W\dot\Gamma_i W$.

\section{Implementation grid}

\paragraph{Complete all-ordinal.}
ULS, DWLS, and WLS are implemented for complete integer data. ULS is covered by
Corollary~\ref{cor:fixed}. DWLS uses the diagonal formula above. Full WLS uses
the full $\dot\Gamma_i$ matrix in the dense inverse-weight derivative. The dense
data-direct channel is gated against case-weight finite differences of the full
NACOV matrix; its diagonal extraction is also checked against the DWLS helper.
A deterministic resampling-jackknife fixture and misspecification simulation
remain useful higher-level checks.

\paragraph{Complete mixed ordinal/polyserial.}
The same equations apply, but $g_i$ and $\dot\Gamma_i$ must cover continuous
means, continuous covariances, ordinal thresholds, polychorics, and polyserials
in the exact mixed moment order. Fixed-weight ULS is implemented: it only needs
casewise $g_i$ and reduces to the observed-bread sandwich. DWLS is also
implemented for ordinary complete-data ML/polyserial mixed stats: the diagonal
$\dot\Gamma_i$ combines the data-direct sandwich channel at fixed mixed kappa
with a finite-difference $\partial\diag(\hat\Gamma)/\partial\kappa$ channel for
thresholds, negative means, variances, polychorics, polyserial covariances, and
Pearson covariance rows. Full WLS uses the dense mixed $\dot\Gamma_i$ with the
same data-direct and finite-difference channels, and its diagonal extraction is
checked against the DWLS helper.

\paragraph{Complete continuous LS.}
For ULS with a fixed identity weight, the IJ is only a wrapper around the
observed-bread sandwich using raw moment influence rows. This slice has landed
as the fixed-weight continuous-LS IJ helper. The sample-built GLS
normal-theory-weight slice has also landed: this is a moment-quadratic
estimated-weight correction, not ordinary ML/FIML robust-score normal-theory
inference. The complete-data WLS/ADF empirical-weight slice has also landed,
using \eqref{eq:continuous-adf-gamma-if} for the full dense
$\dot\Gamma_i$ and a finite-difference contamination gate for both
covariance-only and meanstructure layouts. Complete-data DWLS has landed as
the diagonal extraction of the same empirical-Gamma influence, gated by
case-weight finite differences of $\diag(\hat\Gamma)^{-1}$ in covariance-only
and meanstructure layouts. Complete-data DLS with fixed mixing scalar has also
landed as the convex mixture of the sample-built normal-theory and dense
empirical-Gamma influence channels, with finite-difference gates on the mixed
inverse weight in covariance-only and meanstructure layouts. The raw
mean/covariance $g_i$ is analytic; EB-DLS scalar-selection uncertainty remains
separate from the fixed-$a$ weight derivative. The continuous-LS observed bread
now uses the analytic Hessian in \eqref{eq:continuous-bread}; the public
finite-difference bread helper is retained as the regression oracle for the IJ
adapters.

\paragraph{Two-stage FIML (ML2S).}
Stage 1 supplies saturated EM moments and their ACOV. The observed-bread
single-model regime has landed as \texttt{TwoStageBread::Observed}: it changes
the Stage-2 bread while leaving the Stage-1 ACOV and the current fixed-weight
robust sandwich unchanged. Complete-data gates reduce the NT path to
unstructured observed-bread robust SEM and the ADF path to observed-bread
\texttt{robust\_continuous\_ls}. The casewise saturated-moment influence under
the missingness pattern has also landed as
\texttt{saturated\_em\_moment\_influence}. Let
$s_i(\hat\eta_S)=\partial \ell_i(\eta)/\partial\eta|_{\hat\eta_S}$ be the
observed-pattern saturated log-likelihood score for
$\eta=(\mu,\vech\Sigma)$, and let
$I_\eta=-\sum_i \partial^2\ell_i(\eta)/\partial\eta\partial\eta\T|_{\hat\eta_S}$.
The exported row is
\begin{equation}
  a_i\T = s_i(\hat\eta_S)\T I_\eta^{-1},
  \qquad
  A\T A = I_\eta^{-1}\Bigl(\sum_i s_i s_i\T\Bigr)I_\eta^{-1}
        = \widehat{\mathrm{ACOV}}(\hat\eta_S).
\end{equation}
For complete data this reduces blockwise to
\[
  a_i = n_b^{-1}\bigl[x_i-\bar x_b,\,
        \vech\{(x_i-\bar x_b)(x_i-\bar x_b)\T-\hat\Sigma_b\}\bigr].
\]
For complete raw data, the raw-data \path{two_stage_em_ml_inference}
overload now delegates observed-bread \texttt{Dwls}, \texttt{Adf}, and
fixed-$a$ \texttt{Dls} covariance to the already-gated continuous-LS IJ
adapters; this is an exact reduction because saturated EM moments reduce to
ordinary mean/covariance moments and $n_bA_b\T A_b$ reduces to Browne's complete
data $\hat\Gamma$. The scaled-test fields remain the fixed-weight
Satorra--Bentler quantities because Hall--Inoue test-statistic corrections are
estimator-specific and have not been derived here.

For missing data, the non-NT ML2S IJ adapter cannot just reuse $a_i$ because the
\texttt{Dwls}, \texttt{Adf}, and \texttt{Dls} weights depend on the Stage-1
sandwich meat.  For a block, write
\[
  A = n_b^{-1}H,\qquad
  B = n_b^{-1}J,\qquad
  \Gamma_{\mathrm{FIML}} = A^{-1}BA^{-1}.
\]
If $s_i$ is the H1 loglikelihood score row, $h_i$ is its observed-information
contribution, and $z_i = n_b a_i = A^{-1}s_i$ is the influence row for
$\hat\eta_S$, then the first-order case influence of the FIML meat is
\begin{align}
  \dot\Gamma_i
    &= A^{-1}(s_is_i\T - B)A^{-1}
       - A^{-1}(h_i-A)\Gamma_{\mathrm{FIML}}
       - \Gamma_{\mathrm{FIML}}(h_i-A)A^{-1}
       + D_\eta\Gamma_{\mathrm{FIML}}[z_i].
\end{align}
The four terms are the score-product channel, the observed-information channel,
its transpose, and the movement of $(A,B)$ through the saturated EM solution.
The implementation landed 2026-06-20 takes the conservative route and evaluates
this object by a central case-weight derivative:
\[
  \dot\Gamma_i
    \approx n_b\,
      \frac{\Gamma_{\mathrm{FIML}}(w_i=1+\epsilon)
            -\Gamma_{\mathrm{FIML}}(w_i=1-\epsilon)}
           {2\epsilon},
\]
where each perturbed evaluation reruns the weighted saturated EM update and
rebuilds the weighted $(H,J,H^{-1}JH^{-1})$ stack.  This finite-difference
derivative is slower than the closed-form expansion above, but it includes all
channels and gives a validation target for any later analytic replacement.

The estimated-weight correction row used by
\path{robust_weighted_moment_ij} is
\[
  c_i = r\T W\,\dot\Gamma_i\,W,
\]
with $\dot\Gamma_i$ replaced by only its diagonal for \texttt{Dwls}; for
\texttt{Dls}, $\dot\Gamma_i$ is mixed as
$(1-a)\dot\Gamma_{\mathrm{NT},i}+a\dot\Gamma_{\mathrm{FIML},i}$ before the two
outer $W$ factors.  The default \texttt{TwoStageWeight::Nt} route in
\path{two_stage_em_ml_inference} remains the normal-theory ML robust-score
path; the moment-quadratic GLS normal-theory-weight IJ correction remains
available as \path{robust_continuous_ls_gls_ij} for complete continuous LS
rather than being silently substituted for ML2S NT.

\paragraph{MCAR and pairwise-missing ordinal/polyserial.}
The current primitive assumes disjoint blocks: it sums $V_b\T V_b$ inside each
block. Pairwise missingness has overlapping supports. The correct object is one
case-aligned sparse row vector over the full moment stack, with zeros for
unobserved moment components and the right $N/n_{jk}$ scaling for each
pair-local estimator. The all-ordinal overlap-Gamma slice has landed:
\path{ordinal_stats_from_observed_integer_data(..., Overlap)} materializes
these rows, carries the missing-coded integer block, and gates
\[
  N^{-1}\sum_i g_i g_i\T = \widehat\Gamma_{\mathrm{overlap}} .
\]
Because ULS has $\dot W_i=0$, \path{robust_ordinal_ij} consumes the rows
directly and reduces to the observed-bread fixed-weight sandwich under MCAR or
any pairwise-deletion pattern. For DWLS and WLS, the observed Gamma helpers
reuse the same missing pattern at fixed $\kappa$: unsupported score components
and case-bread entries are zero, while observed threshold and pairwise
polychoric pieces contribute to $\dot V_i$, $\dot A_i$, and the
finite-difference $\partial\Gamma/\partial\kappa$ channel. The nominal-Gamma
variant is an elementwise support rescaling and is therefore not represented as
casewise IF rows. Mixed ordinal/polyserial MCAR still needs the same
support-aligned row and $\dot\Gamma_i$ construction; the remaining gates are
case-weight finite differences for $\dot\Gamma_i$, MCAR simulation against a
case-resampling jackknife, and equality of pairwise-overlap covariance to the
existing NACOV.

\section{Verification standard}

Every adapter gets the same four checks before it is trusted:
\begin{enumerate}
\item fixed-weight reduction to \texttt{robust\_weighted\_moments};
\item finite-difference or case-weight derivative check for each new
  $\dot\Gamma_i$ channel;
\item resampling-jackknife agreement on a small deterministic fixture;
\item simulation calibration under the misspecified model that motivated the
  method, with ULS/fixed-weight cells used as the negative control.
\end{enumerate}
